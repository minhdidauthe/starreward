Code Examples
Basic Usage
// Simple toasts
Toast.success('Title', 'Message');
Toast.error('Title', 'Message');
Toast.warning('Title', 'Message');
Toast.info('Title', 'Message');

// With options
Toast.success('Title', 'Message', {
    duration: 5000,
    filled: true
});
Loading & Promise
// Loading toast
const loader = Toast.loading('Đang xử lý...');
// Later...
loader.success('Hoàn thành!');
// or
loader.error('Thất bại!');

// Promise toast
Toast.promise(fetch('/api/data'), {
    loading: { title: 'Đang tải...' },
    success: { title: 'Thành công!' },
    error: { title: 'Lỗi!' }
});
Action Toast
Toast.action('Đã xóa', 'File đã bị xóa', {
    label: 'Hoàn tác',
    cancel: 'Đóng',
    onClick: () => {
        // Undo action
        console.log('Undoing...');
    }
});
Position & Settings
// Change position
Toast.setPosition('top-right');
// Options: bottom-center, top-center,
//          top-right, bottom-right

// Dismiss all
Toast.dismissAll();

// Initialize with options
Toast.init({
    position: 'top-center',
    maxToasts: 3
});
